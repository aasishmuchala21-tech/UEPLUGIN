**CameraMoveTaxonomy enum** (canonical taxonomy per ROADMAP SC#3):
```python
class CameraMoveType(str, Enum):
    STATIC = "static"          # Camera does not move
    PAN = "pan"                # Horizontal rotation around a fixed point
    TILT = "tilt"              # Vertical rotation around a fixed point
    DOLLY = "dolly"            # Camera moves along Z-axis (toward/away from subject)
    TRUCK = "truck"            # Camera moves along X or Y axis (lateral, parallel to frame)
    CRANE = "crane"            # Camera moves on multiple axes (complex, dolly + truck + tilt combined)
    ZOOM = "zoom"              # FOV change only, camera does not move physically
    HANDHELD = "handheld"      # Organic jitter — simulated with noise, not scripted keyframes
    UNKNOWN = "unknown"        # Requires user confirmation
```

**SequencerToolMixin** (shared Sequencer API helpers):
```python
class SequencerToolMixin:
    """Mix-in providing Unreal Sequencer scripting API helpers.

    Uses unreal.get_editor_subsystem(unreal.LevelSequenceEditorSubsystem) for
    Sequencer binding and keyframe authoring. Falls back to unreal.MovieSceneAPI
    for lower-level access when the subsystem is not available.
    """

    def _get_sequence_editor(self) -> Any:
        return unreal.get_editor_subsystem(unreal.LevelSequenceEditorSubsystem)

    def _create_level_sequence(self, sequence_name: str, level: Any = None) -> Any:
        """Create a new ULevelSequence in the current level.

        Uses unreal.LevelSequenceEditorBlueprintLibrary.create_level_sequence().
        Returns the created ULevelSequence actor.
        """
        seq = unreal.LevelSequenceEditorBlueprintLibrary.create_level_sequence(
            sequence_name
        )
        return seq

    def _bind_camera_to_sequence(self, sequence: Any, camera_actor: Any) -> Any:
        """Bind a CineCamera or CameraActor to the level sequence.

        Adds the camera as a possessable object in the sequence and returns the
        movie scene binding object for the camera.
        """
        subystem = self._get_sequence_editor()
        binding = subystem.add_possible_binding_to_sequence(
            sequence, camera_actor
        )
        return binding

    def _set_transform_keyframe(
        self,
        sequence: Any,
        binding: Any,
        frame_number: int,
        location: tuple[float, float, float],
        rotation: tuple[float, float, float],
        scale: tuple[float, float, float] = (1.0, 1.0, 1.0),
    ) -> None:
        """Set a transform keyframe (location + rotation) on a binding."""
        # Convert seconds to frame number (assume 24fps)
        # Use MovieSceneTrackEditorUtilities or SequencerScripting library
        # Keyframe on CameraCut track or the camera component's transform track
        pos = unreal.Vector(location[0], location[1], location[2])
        rot = unreal.Rotator(rotation[0], rotation[1], rotation[2])
        sc = unreal.Vector(scale[0], scale[1], scale[2])
        # The actual keyframe setting uses the Sequencer scripting API
        # unreal.LevelSequenceEditorBlueprintLibrary.add_keyframe_*_transform()
        unreal.LevelSequenceEditorBlueprintLibrary.add_keyframe_absolute_transform(
            binding, sequence, pos, rot, sc, float(frame_number)
        )

    def _set_camera_fov_keyframe(
        self,
        sequence: Any,
        binding: Any,
        frame_number: int,
        fov_degrees: float,
    ) -> None:
        """Set a FOV keyframe on a CineCameraComponent."""
        unreal.LevelSequenceEditorBlueprintLibrary.add_keyframe_absolute_focal_focus(
            binding, sequence, fov_degrees, float(frame_number)
        )

    def _set_light_intensity_keyframe(
        self,
        sequence: Any,
        binding: Any,
        frame_number: int,
        intensity: float,
    ) -> None:
        """Set a light intensity keyframe."""
        unreal.LevelSequenceEditorBlueprintLibrary.add_keyframe_absolute_light_intensity(
            binding, sequence, intensity, float(frame_number)
        )

    def _set_light_color_keyframe(
        self,
        sequence: Any,
        binding: Any,
        frame_number: int,
        color_rgb: tuple[float, float, float],
    ) -> None:
        """Set a light color keyframe."""
        color = unreal.LinearColor(color_rgb[0], color_rgb[1], color_rgb[2], 1.0)
        unreal.LevelSequenceEditorBlueprintLibrary.add_keyframe_absolute_light_color(
            binding, sequence, color, float(frame_number)
        )
```

```python
        "type": "object",
        "properties": {
            "sequence_path": {
                "type": "string",
                "description": "Path to the ULevelSequence asset",
            },
            "camera_name": {
                "type": "string",
                "description": "Label for the camera actor in the World Outliner",
                "default": "NYRA_SequencerCam",
            },
            "location": {
                "type": "object",
                "properties": {"x": {"type": "number"}, "y": {"type": "number"}, "z": {"type": "number"}},
                "default": {"x": 0.0, "y": 0.0, "z": 100.0},
            },
            "fov": {
                "type": "number",
                "default": 35.0,
                "description": "Horizontal FOV in degrees (35mm equivalent)",
            },
            "focus_distance": {
                "type": "number",
                "default": 3.0,
                "description": "Focus distance in meters",
            },
        },
        "required": ["sequence_path"],
    }

    def execute(self, params: dict) -> NyraToolResult:
        # 1. Spawn CineCameraActor at location
        camera_tool = ActorSpawnTool()
        spawn_result = camera_tool.execute({
            "class_name": "CineCameraActor",
            "name": params.get("camera_name", "NYRA_SequencerCam"),
            "location": params.get("location", {"x": 0.0, "y": 0.0, "z": 100.0}),
        })
        if not spawn_result.is_ok:
            return NyraToolResult.err(f"Failed to spawn CineCamera: {spawn_result.error}")

        # 2. Load sequence
        seq = unreal.EditorAssetLibrary.load_asset(params["sequence_path"])
        if seq is None:
            return NyraToolResult.err(f"Level sequence not found: {params['sequence_path']}")

        # 3. Bind camera to sequence
        binding = self._bind_camera_to_sequence(seq, spawn_result.data["actor_path"])
        if binding is None:
            return NyraToolResult.err(f"Failed to bind camera to sequence")

        # 4. Set initial FOV + focus on the CineCameraComponent
        camera_actor = unreal.EditorLevelLibrary.get_actor_reference(spawn_result.data["actor_path"])
        camera_comp = camera_actor.get_cine_camera_component()
        if camera_comp:
            camera_comp.set_field_of_view(params.get("fov", 35.0))
            camera_comp.set_focus_distance(params.get("focus_distance", 3.0) * 100.0)  # cm

        log.info("camera_added_to_sequence", camera=spawn_result.data["actor_name"],
                 sequence=params["sequence_path"])
        return NyraToolResult.ok({
            "camera_actor_name": spawn_result.data["actor_name"],
            "camera_actor_path": spawn_result.data["actor_path"],
            "camera_actor_guid": spawn_result.data["guid"],
            "sequence_path": params["sequence_path"],
            "fov": params.get("fov", 35.0),
            "focus_distance_m": params.get("focus_distance", 3.0),
            "message": f"CineCamera '{spawn_result.data['actor_name']}' added and bound to sequence",
        })
```

```python
class SequencerSetKeyframeTool(NyraTool):
    name = "nyra_sequencer_set_keyframe"
    description = (
        "Set a keyframe on a CineCamera or light in a level sequence. "
        "Sets transform (location + rotation), FOV, focus, light intensity, or light color. "
        "Time is specified in seconds (24fps assumed)."
    )
    parameters = {
        "type": "object",
        "properties": {
            "sequence_path": {"type": "string"},
            "binding_path": {
                "type": "string",
                "description": "Actor path (world outliner) of the bound camera or light",
            },
            "time_seconds": {"type": "number", "description": "Time in seconds"},
            "transform": {
                "type": "object",
                "properties": {
                    "location": {"x": {"type": "number"}, "y": {"type": "number"}, "z": {"type": "number"}},
                    "rotation": {"pitch": {"type": "number"}, "yaw": {"type": "number"}, "roll": {"type": "number"}},
                },
            },
            "fov_degrees": {"type": "number"},
            "focus_distance_m": {"type": "number"},
            "light_intensity": {"type": "number"},
            "light_color": {"type": "array", "items": {"type": "number"}, "description": "RGB 0-1"},
        },
        "required": ["sequence_path", "binding_path", "time_seconds"],
    }

    def execute(self, params: dict) -> NyraToolResult:
        seq = unreal.EditorAssetLibrary.load_asset(params["sequence_path"])
        if seq is None:
            return NyraToolResult.err(f"Level sequence not found: {params['sequence_path']}")

        # Get the binding for this actor in the sequence
        binding = self._find_binding_for_actor(seq, params["binding_path"])
        if binding is None:
            return NyraToolResult.err(f"Actor not bound to sequence: {params['binding_path']}")

        frame = int(params["time_seconds"] * 24.0)  # 24fps

        if params.get("transform"):
            t = params["transform"]
            loc = (t["location"]["x"], t["location"]["y"], t["location"]["z"])
            rot = (t["rotation"]["pitch"], t["rotation"]["yaw"], t["rotation"]["roll"])
            self._set_transform_keyframe(seq, binding, frame, loc, rot)

        if params.get("fov_degrees"):
            self._set_camera_fov_keyframe(seq, binding, frame, params["fov_degrees"])

        if params.get("light_intensity"):
            self._set_light_intensity_keyframe(seq, binding, frame, params["light_intensity"])

        if params.get("light_color"):
            self._set_light_color_keyframe(seq, binding, frame, tuple(params["light_color"]))

        log.info("keyframe_set", binding=params["binding_path"], time_s=params["time_seconds"],
                frame=frame)
        return NyraToolResult.ok({
            "status": "keyframe_set",
            "sequence_path": params["sequence_path"],
            "binding_path": params["binding_path"],
            "time_seconds": params["time_seconds"],
            "frame": frame,
            "fields_set": list(params.keys()),
        })

    def _find_binding_for_actor(self, sequence: Any, actor_path: str) -> Any:
        """Find the movie scene binding for an actor in the sequence."""
        # Use LevelSequenceEditorBlueprintLibrary.find_binding_by_id or iterate bindings
        # Return the first binding whose actor path matches
        editor = self._get_sequence_editor()
        # Sequencer bindings are identified by GUID; find by iterating
        bindings = editor.get_possible_bindings(sequence)
        for binding in bindings:
            if binding.get_name() == actor_path or actor_path in binding.get_name():
                return binding
        return None
```

```python
        start = params.get("start_time_seconds", 0.0)
        end = start + duration

        if params.get("video_reference_json"):
            ref = VideoReferenceParams.from_json(params["video_reference_json"])
            return self._author_from_video_reference(seq, binding, ref, start, end, params)
        elif params.get("nl_description"):
            return self._author_from_nl(seq, binding, params["nl_description"], start, end, params)
        else:
            return NyraToolResult.err("[-32030] Either video_reference_json or nl_description required")

    def _author_from_video_reference(
        self, seq: Any, binding: Any, ref: VideoReferenceParams,
        start: float, end: float, params: dict
    ) -> NyraToolResult:
        """Convert VideoReferenceParams to keyframes."""
        for shot in ref.shot_blocks:
            if not shot.user_confirmed and shot.camera_move_type == CameraMoveType.UNKNOWN:
                log.warning("shot_not_confirmed", shot_id=shot.shot_id)
                continue  # skip unconfirmed shots

            # Convert CameraMoveType to keyframe pattern
            pattern = self._camera_move_to_keyframe_pattern(shot, start, end)
            for keyframe in pattern:
                frame = int(keyframe["time"] * 24.0)
                self._set_transform_keyframe(
                    seq, binding, frame,
                    keyframe["location"], keyframe["rotation"]
                )
                if keyframe.get("fov"):
                    self._set_camera_fov_keyframe(seq, binding, frame, keyframe["fov"])

        return NyraToolResult.ok({
            "status": "shot_authored",
            "video_reference": True,
            "shot_count": len(ref.shot_blocks),
            "start_time": start,
            "end_time": end,
            "camera_move": ref.camera_move_type,
        })

    def _author_from_nl(
        self, seq: Any, binding: Any, nl_desc: str,
        start: float, end: float, params: dict
    ) -> NyraToolResult:
        """Parse NL shot description and convert to keyframes.

        Rules:
        - "slow push-in" / "dolly in" / "move in" -> dolly: move camera Z forward
        - "cut wide" / "pull back" -> dolly: move camera Z back
        - "pan left/right" -> truck: lateral X movement
        - "tilt up/down" -> tilt: rotation pitch change
        - "static" / "locked off" -> no keyframes (camera holds position)
        - "handheld" -> add small noise to keyframes (random offsets within 5cm)
        """
        desc = nl_desc.lower()
        camera_move = CameraMoveType.UNKNOWN
        keyframes = []

        if "push-in" in desc or "dolly in" in desc or "move in" in desc or "crane in" in desc:
            camera_move = CameraMoveType.DOLLY
            # 2 keyframes: start + end
            keyframes = [
                {"time": start, "location": (0, 0, 100), "rotation": (0, 0, 0)},
                {"time": end, "location": (0, 0, 50), "rotation": (0, 0, 0)},
            ]
        elif "cut wide" in desc or "pull back" in desc or "dolly out" in desc:
            camera_move = CameraMoveType.DOLLY
            keyframes = [
                {"time": start, "location": (0, 0, 50), "rotation": (0, 0, 0)},
                {"time": end, "location": (0, 0, 100), "rotation": (0, 0, 0)},
            ]
        elif "pan left" in desc or "track left" in desc:
            camera_move = CameraMoveType.TRUCK
            keyframes = [
                {"time": start, "location": (0, 0, 100), "rotation": (0, 0, 0)},
                {"time": end, "location": (-200, 0, 100), "rotation": (0, 0, 0)},
            ]
        elif "pan right" in desc or "track right" in desc:
            camera_move = CameraMoveType.TRUCK
            keyframes = [
                {"time": start, "location": (0, 0, 100), "rotation": (0, 0, 0)},
                {"time": end, "location": (200, 0, 100), "rotation": (0, 0, 0)},
            ]
        elif "static" in desc or "locked off" in desc:
            camera_move = CameraMoveType.STATIC
            keyframes = [
                {"time": start, "location": (0, 0, 100), "rotation": (0, 0, 0)},
                {"time": end, "location": (0, 0, 100), "rotation": (0, 0, 0)},
            ]
        elif "tilt up" in desc:
            camera_move = CameraMoveType.TILT
            keyframes = [
                {"time": start, "location": (0, 0, 100), "rotation": (-10, 0, 0)},
                {"time": end, "location": (0, 0, 100), "rotation": (10, 0, 0)},
            ]
        elif "tilt down" in desc:
            camera_move = CameraMoveType.TILT
            keyframes = [
                {"time": start, "location": (0, 0, 100), "rotation": (10, 0, 0)},
                {"time": end, "location": (0, 0, 100), "rotation": (-10, 0, 0)},
            ]
        else:
            # Unknown motion — default to slow dolly-in
            camera_move = CameraMoveType.UNKNOWN
            keyframes = [
                {"time": start, "location": (0, 0, 100), "rotation": (0, 0, 0)},
                {"time": end, "location": (0, 0, 75), "rotation": (0, 0, 0)},
            ]

        for kf in keyframes:
            frame = int(kf["time"] * 24.0)
            self._set_transform_keyframe(
                seq, binding, frame,
                kf["location"], kf["rotation"]
            )

        log.info("shot_authored_from_nl", nl_description=nl_desc,
                 camera_move=camera_move, keyframe_count=len(keyframes))
        return NyraToolResult.ok({
            "status": "shot_authored",
            "nl_description": nl_desc,
            "camera_move_type": camera_move,
            "keyframe_count": len(keyframes),
            "start_time": start,
            "end_time": end,
        })

    def _camera_move_to_keyframe_pattern(
        self, shot: ShotBlock, start: float, end: float
    ) -> list[dict]:
        """Convert a ShotBlock to keyframe dicts based on CameraMoveType."""
        move_type = shot.camera_move_type
        # User override
        if shot.user_override_move_type:
            move_type = shot.user_override_move_type

        # 4 keyframes spread across shot duration for smooth curves
        times = [start, start + (end - start) / 3, start + 2 * (end - start) / 3, end]

        if move_type == CameraMoveType.STATIC:
            base_loc = shot.start_position
            return [{"time": t, "location": base_loc, "rotation": shot.start_rotation}
                    for t in times]
        elif move_type == CameraMoveType.DOLLY:
            start_loc = shot.start_position
            end_loc = shot.end_position
            return [{"time": t, "location": start_loc, "rotation": shot.start_rotation}
                    for t in [start, end]]
        elif move_type == CameraMoveType.TRUCK:
            start_loc = shot.start_position
            end_loc = shot.end_position
            return [{"time": t, "location": start_loc, "rotation": shot.start_rotation}
                    for t in [start, end]]
        else:
            # Generic: just set start + end keyframes
            return [
                {"time": start, "location": shot.start_position, "rotation": shot.start_rotation},
                {"time": end, "location": shot.end_position, "rotation": shot.end_rotation},
            ]

    def _find_binding_for_actor(self, sequence: Any, actor_path: str) -> Any:
        editor = self._get_sequence_editor()
        bindings = editor.get_possible_bindings(sequence)
        for binding in bindings:
            if binding.get_name() == actor_path or actor_path in binding.get_name():
                return binding
        return None
```
