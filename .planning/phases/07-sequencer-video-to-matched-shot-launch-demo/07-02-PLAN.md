"""nyrahost.tools.shot_block_ui — User-confirmable camera-move taxonomy.

Phase 7 Wave 2: Provides shot confirmation UI logic for the WS-triggered Slate
widget, and the camera-move override mechanism for dolly/truck confusion
(per ROADMAP SC#3 PITFALLS).
"""
from __future__ import annotations

import structlog
from dataclasses import replace
from typing import Optional

from nyrahost.tools.video_llm_parser import CameraMoveType, ShotBlock, VideoReferenceParams

log = structlog.get_logger("nyrahost.tools.shot_block_ui")

__all__ = ["ShotBlockConfirmationUI"]

# Camera move type display names
CAMERA_MOVE_DISPLAY = {
    CameraMoveType.STATIC: "Locked Off (Static)",
    CameraMoveType.PAN: "Pan (horizontal rotation)",
    CameraMoveType.TILT: "Tilt (vertical rotation)",
    CameraMoveType.DOLLY: "Dolly (move toward/away)",
    CameraMoveType.TRUCK: "Truck (lateral pan)",
    CameraMoveType.CRANE: "Crane (multi-axis)",
    CameraMoveType.ZOOM: "Zoom (FOV change)",
    CameraMoveType.HANDHELD: "Handheld (organic jitter)",
    CameraMoveType.UNKNOWN: "Unknown (needs confirmation)",
}

# Common confusion pairs
CONFUSION_PAIRS = {
    CameraMoveType.DOLLY: CameraMoveType.TRUCK,
    CameraMoveType.TRUCK: CameraMoveType.DOLLY,
}


class ShotBlockConfirmationUI:
    """Handles shot block confirmation and camera-move override logic.

    The actual Slate UI (SMyrShotConfirmDialog) lives in the C++ NyraEditor
    module and calls into this Python module via WS messages. This class provides
    the data formatting and override logic that the UI needs.

    Per ROADMAP SC#3: camera-move taxonomy is user-confirmable.
    Per PITFALLS §6.2: dolly vs truck confusion is a known risk — users can override.
    """

    def __init__(self, confusion_pairs: Optional[dict] = None):
        """Initialize with optional confusion pair overrides."""
        self.confusion_pairs = confusion_pairs or CONFUSION_PAIRS

    def _format_confirmation_message(self, shot: ShotBlock) -> str:
        """Format a human-readable shot confirmation message.

        Args:
            shot: The ShotBlock to format

        Returns:
            Multiline message string for the confirmation dialog
        """
        display_type = CAMERA_MOVE_DISPLAY.get(shot.camera_move_type, "Unknown")

        lines = [
            f"Shot: {shot.shot_id}",
            f"Camera movement: {display_type}",
            f"FOV: {shot.fov:.0f}mm equivalent",
            f"Focus: {shot.focus_distance:.1f}m / f{shot.aperture}",
            f"Description: {shot.nl_description}",
            f"Duration: {shot.end_time - shot.start_time:.1f}s",
        ]

        if shot.camera_move_type in self.confusion_pairs:
            override_type = self.confusion_pairs[shot.camera_move_type]
            override_display = CAMERA_MOVE_DISPLAY.get(override_type, str(override_type))
            lines.append("")
            lines.append(
                f"[NOTE] This could also be: {override_display}. "
                "Use 'Override' if the camera moves sideways (truck) rather than forward (dolly)."
            )

        return "\n".join(lines)

    def confirm_shot(
        self,
        shot: ShotBlock,
        override_move_type: Optional[CameraMoveType] = None,
        override_confirmed: bool = True,
    ) -> ShotBlock:
        """Mark a shot as confirmed by the user.

        Args:
            shot: The ShotBlock from video analysis
            override_move_type: If user selects a different camera move type
            override_confirmed: Always True (user clicked confirm)

        Returns:
            New ShotBlock with user_confirmed=True and optional override
        """
        if override_move_type is not None:
            confirmed = replace(
                shot,
                user_confirmed=override_confirmed,
                user_override_move_type=override_move_type,
            )
            log.info("shot_confirmed_with_override",
                     shot_id=shot.shot_id,
                     original_move=shot.camera_move_type.value,
                     override_move=override_move_type.value)
        else:
            confirmed = replace(shot, user_confirmed=override_confirmed)
            log.info("shot_confirmed",
                     shot_id=shot.shot_id,
                     camera_move=shot.camera_move_type.value)

        return confirmed

    def format_confirmation_card(self, params: VideoReferenceParams) -> dict:
        """Format a full shot confirmation card from VideoReferenceParams.

        Args:
            params: VideoReferenceParams from video analysis

        Returns:
            Dict with formatted fields for the Slate confirmation UI
        """
        primary_shot = params.shot_blocks[0] if params.shot_blocks else None

        return {
            "camera_move_type": params.camera_move_type.value,
            "camera_move_display": CAMERA_MOVE_DISPLAY.get(
                params.camera_move_type, "Unknown"
            ),
            "confidence": params.camera_move_confidence,
            "framing": params.framing,
            "subject_position": list(params.subject_position),
            "rule_of_thirds": params.rule_of_thirds,
            "lighting_mood_tags": params.lighting_mood_tags,
            "lighting_temperature_k": params.primary_temperature_k,
            "environment_type": params.environment_type,
            "time_of_day": params.time_of_day,
            "weather": params.weather,
            "geometry_categories": params.geometry_categories,
            "duration_s": params.clip_duration_seconds,
            "keyframe_count": params.keyframe_count,
            "analysis_confidence": params.analysis_confidence,
            "shot_count": len(params.shot_blocks),
            "primary_shot_nl_description": primary_shot.nl_description if primary_shot else "",
            "requires_override": params.camera_move_type in self.confusion_pairs,
            "possible_override": CAMERA_MOVE_DISPLAY.get(
                self.confusion_pairs.get(params.camera_move_type, CameraMoveType.UNKNOWN),
                ""
            ),
            "confusion_note": (
                "NYRA detected this as DOLLY. If the camera moves sideways (lateral), "
                "select TRUCK instead."
                if params.camera_move_type == CameraMoveType.DOLLY else
                "NYRA detected this as TRUCK. If the camera moves forward/back, "
                "select DOLLY instead."
                if params.camera_move_type == CameraMoveType.TRUCK else
                ""
            ),
        }

"""nyrahost.tools.demo02_orchestrator — DEMO-02 end-to-end orchestration.

Phase 7 Wave 2-3: Orchestrates the full DEMO-02 pipeline:
  1. VideoReferenceAnalyzer.analyze() -> VideoReferenceParams
  2. Check requires_user_confirmation()
  3a. If True: return confirmation card data + requires_confirmation=True
  3b. If False: auto-commit to _assemble_scene()
  4. _assemble_scene(): lighting -> CineCamera -> Sequencer camera

This is the top-level orchestrator that the MCP tool calls (nyra_demo02_run).
"""
from __future__ import annotations

import asyncio
import structlog
from dataclasses import replace
from typing import Any, Optional

from nyrahost.tools.base import NyraTool, NyraToolResult
from nyrahost.tools.ffmpeg_extractor import FFmpegKeyframeExtractor
from nyrahost.tools.lighting_tools import LightingAuthoringTool, LightingParams
from nyrahost.tools.sequencer_tools import (
    SequencerAddCameraTool,
    SequencerAuthorShotTool,
    SequencerCreateTool,
    SequencerSetKeyframeTool,
)
from nyrahost.tools.shot_block_ui import ShotBlockConfirmationUI
from nyrahost.tools.video_llm_parser import CameraMoveType, ShotBlock, VideoReferenceParams
from nyrahost.tools.video_reference_analyzer import VideoReferenceAnalyzer

log = structlog.get_logger("nyrahost.tools.demo02_orchestrator")

__all__ = ["Demo02Orchestrator", "Demo02Tool"]


# Scene scale constants (world units = cm)
SCENE_SCALE_X = 500.0  # Scene is 5m wide
SCENE_SCALE_Z = 100.0  # Default camera height 1m


class Demo02Orchestrator:
    """End-to-end DEMO-02 orchestrator.

    Pipeline:
      analyze(source, source_type) -> VideoReferenceParams
        -> requires_confirmation? -> return confirmation card
        -> else: assemble_scene() -> return result

      assemble_scene(params, confirmed_shot_blocks) -> orchestrator result
        -> LightingAuthoringTool (from SCENE-01)
        -> SequencerCreateTool -> SequencerAddCameraTool
        -> SequencerAuthorShotTool (with confirmed shot blocks)
    """

    def __init__(self, claude_backend: Any = None):
        self.claude_backend = claude_backend
        self.analyzer = VideoReferenceAnalyzer(claude_backend=claude_backend)
        self.confirmation_ui = ShotBlockConfirmationUI()
        # Tool instances for assembly
        self._lighting_tool = LightingAuthoringTool()
        self._sequencer_create = SequencerCreateTool()
        self._sequencer_add_camera = SequencerAddCameraTool()
        self._sequencer_set_keyframe = SequencerSetKeyframeTool()
        self._sequencer_author_shot = SequencerAuthorShotTool()

    async def analyze_and_assemble(
        self,
        source: str,
        source_type: str = "file",
        confirmed_shot_blocks: Optional[list[ShotBlock]] = None,
    ) -> dict[str, Any]:
        """Run the full DEMO-02 pipeline.

        Args:
            source: YouTube URL or local file path
            source_type: "youtube_url" or "file"
            confirmed_shot_blocks: If requires_confirmation=True, user-confirmed shot blocks

        Returns:
            Dict with:
              - requires_confirmation: bool
              - confirmation_card: dict (if requires_confirmation=True)
              - sequence_path: str (after assembly)
              - camera_actor_path: str
              - lighting_mood_tags: list[str]
        """
        # Step 1: Analyze video
        params = await self._analyze_video(source, source_type)

        # Step 2: Check if user confirmation is needed
        if params.requires_user_confirmation() and not confirmed_shot_blocks:
            confirmation_card = self.confirmation_ui.format_confirmation_card(params)
            log.info("demo02_requires_confirmation",
                    confidence=params.camera_move_confidence,
                    camera_move=params.camera_move_type.value)
            return {
                "requires_confirmation": True,
                "confirmation_card": confirmation_card,
                "video_reference_params": {
                    "camera_move_type": params.camera_move_type.value,
                    "camera_move_confidence": params.camera_move_confidence,
                    "shot_blocks": [
                        {
                            "shot_id": sb.shot_id,
                            "camera_move_type": sb.camera_move_type.value,
                            "nl_description": sb.nl_description,
                            "fov": sb.fov,
                            "focus_distance": sb.focus_distance,
                        }
                        for sb in params.shot_blocks
                    ],
                },
                "message": (
                    f"I detected a {params.camera_move_type.value} shot at "
                    f"{params.camera_move_confidence:.0%} confidence. "
                    "Before I build the scene, do you want to confirm or adjust the camera movement?"
                ),
            }

        # Step 3: Use confirmed shot blocks (or auto-confirm if high confidence)
        if confirmed_shot_blocks:
            final_shot_blocks = confirmed_shot_blocks
        else:
            # Auto-confirm high-confidence shots
            final_shot_blocks = [
                replace(sb, user_confirmed=True) for sb in params.shot_blocks
            ]

        # Step 4: Assemble the scene
        result = await self._assemble_scene(params, final_shot_blocks)

        log.info("demo02_complete",
                 sequence=result.get("sequence_path"),
                 camera_move=params.camera_move_type.value,
                 lighting_moods=params.lighting_mood_tags)

        return {
            "requires_confirmation": False,
            "sequence_path": result.get("sequence_path"),
            "camera_actor_path": result.get("camera_actor_path"),
            "camera_move_type": params.camera_move_type.value,
            "lighting_mood_tags": params.lighting_mood_tags,
            "framing": params.framing,
            "environment_type": params.environment_type,
            "time_of_day": params.time_of_day,
            "weather": params.weather,
            "geometry_categories": params.geometry_categories,
            "message": (
                f"DEMO-02 complete: {params.camera_move_type.value} shot assembled. "
                f"Level sequence created at {result.get('sequence_path')}. "
                f"Lighting: {', '.join(params.lighting_mood_tags)}. "
                f"Open Sequencer to review and adjust keyframes."
            ),
        }

    async def _analyze_video(self, source: str, source_type: str) -> VideoReferenceParams:
        """Analyze video with VideoReferenceAnalyzer."""
        return await self.analyzer.analyze(source=source, source_type=source_type)

    async def _assemble_scene(
        self, params: VideoReferenceParams, confirmed_shot_blocks: list[ShotBlock]
    ) -> dict[str, Any]:
        """Assemble the UE scene from VideoReferenceParams.

        Pipeline:
          1. Apply lighting (LightingAuthoringTool)
          2. Create level sequence
          3. Spawn CineCamera at derived position
          4. Author shot with confirmed shot blocks
        """
        # 1. Lighting
        lighting_params = params.get_primary_lighting_params()
        if lighting_params.mood_tags:
            # Apply via LightingAuthoringTool (synchronous wrapper)
            self._lighting_tool.execute({
                "apply": True,
                "nl_prompt": ",".join(lighting_params.mood_tags),
            })
            log.info("demo02_lighting_applied", moods=lighting_params.mood_tags)

        # 2. Create level sequence
        sequence_result = self._sequencer_create.execute({
            "sequence_name": f"NYRA_Demo02_{params.environment_type}",
            "duration_seconds": params.clip_duration_seconds,
        })
        if not sequence_result.is_ok:
            raise RuntimeError(f"Failed to create sequence: {sequence_result.error}")
        sequence_path = sequence_result.data["sequence_path"]

        # 3. Spawn CineCamera at position derived from subject_position
        # subject_position is normalized (0-1); map to world space
        # Default: camera at subject_x * SCENE_SCALE_X, SCENE_SCALE_Z (z-height)
        cam_x = params.subject_position[0] * SCENE_SCALE_X
        cam_y = -200  # Behind the subject
        cam_z = SCENE_SCALE_Z

        camera_result = self._sequencer_add_camera.execute({
            "sequence_path": sequence_path,
            "camera_name": "NYRA_Demo02_Camera",
            "location": {"x": cam_x, "y": cam_y, "z": cam_z},
            "fov": confirmed_shot_blocks[0].fov if confirmed_shot_blocks else 35.0,
            "focus_distance": confirmed_shot_blocks[0].focus_distance if confirmed_shot_blocks else 3.0,
        })
        if not camera_result.is_ok:
            raise RuntimeError(f"Failed to add camera: {camera_result.error}")
        camera_actor_path = camera_result.data["camera_actor_path"]

        # 4. Author shot with confirmed shot blocks
        if confirmed_shot_blocks:
            # Serialize confirmed shot blocks to JSON for the tool
            import json
            shot_blocks_json = json.dumps([
                {
                    "shot_id": sb.shot_id,
                    "camera_move_type": (
                        sb.user_override_move_type.value
                        if sb.user_override_move_type else sb.camera_move_type.value
                    ),
                    "start_time": sb.start_time,
                    "end_time": sb.end_time,
                    "start_position": list(sb.start_position),
                    "end_position": list(sb.end_position),
                    "start_rotation": list(sb.start_rotation),
                    "end_rotation": list(sb.end_rotation),
                    "fov": sb.fov,
                    "nl_description": sb.nl_description,
                    "user_confirmed": sb.user_confirmed,
                }
                for sb in confirmed_shot_blocks
            ])

            shot_result = self._sequencer_author_shot.execute({
                "sequence_path": sequence_path,
                "binding_path": camera_actor_path,
                "video_reference_json": shot_blocks_json,
                "duration_seconds": params.clip_duration_seconds,
            })
            if not shot_result.is_ok:
                log.warning("shot_authoring_partial", error=shot_result.error)

        return {
            "sequence_path": sequence_path,
            "camera_actor_path": camera_actor_path,
        }
