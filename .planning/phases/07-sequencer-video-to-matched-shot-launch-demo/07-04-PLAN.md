"""tests/test_demo02_cold_start.py — DEMO-02 cold-start smoke test.

Per ROADMAP SC#5: cold-start test must pass on 3 consecutive random clips.
This test validates the cold-start reliability path:
- No external API keys required (Claude, Meshy, ComfyUI all optional for DEMO-02)
- No Meshy dependency (DEMO-02 uses asset library first)
- No ComfyUI dependency (DEMO-02 generates lighting from reference)
- yt-dlp + ffmpeg bundled in Phase 5

Test strategy: mock Claude backend, simulate random clips, verify pipeline
doesn't block on missing dependencies.
"""
import pytest
from unittest.mock import MagicMock, patch, AsyncMock
from pathlib import Path

import sys
mock_unreal = MagicMock()
sys.modules['unreal'] = mock_unreal


class TestDemo02ColdStart:
    """Cold-start smoke tests for DEMO-02.

    These tests validate that DEMO-02 doesn't require any external service
    to start the pipeline (other than ffmpeg + yt-dlp which are bundled).
    """

    def test_demo02_pipeline_starts_without_api_key(self):
        """DEMO-02 pipeline can start even if Claude API key is not set.

        Note: NyraHost must handle the case where claude_backend is None gracefully.
        For cold-start test: we verify the pipeline checks for claude_backend
        and raises a clear error, rather than silently blocking.
        """
        from nyrahost.tools.demo02_cli_tool import Demo02CLITool
        from nyrahost.tools.demo02_orchestrator import Demo02Orchestrator

        # No backend provided — should raise clear error
        tool = Demo02CLITool(claude_backend=None)
        orchestrator = Demo02Orchestrator(claude_backend=None)

        # The orchestrator should detect no backend and raise clear error
        with pytest.raises(RuntimeError, match="no claude_backend"):
            # This simulates what happens when user tries to run without Claude configured
            async def run():
                return await orchestrator.analyze_and_assemble("/fake.mp4")

            import asyncio
            asyncio.get_event_loop().run_until_complete(run())

    def test_duration_validation_does_not_require_network(self):
        """Duration validation for local files doesn't require network."""
        from nyrahost.tools.demo02_cli_tool import Demo02CLITool

        tool = Demo02CLITool(claude_backend=MagicMock())

        # Auto-detect works without network
        assert tool._detect_source_type("/Users/test/clip.mp4") == "file"
        assert tool._detect_source_type("/Users/test/clip.mov") == "file"
        assert tool._detect_source_type("https://youtube.com/watch?v=test") == "youtube_url"

    def test_ffmpeg_extractor_resolves_from_system_path(self):
        """FFmpegKeyframeExtractor falls back to system PATH when bundled not found."""
        from nyrahost.tools.ffmpeg_extractor import FFmpegKeyframeExtractor

        extractor = FFmpegKeyframeExtractor(ffmpeg_path=None)
        # Should resolve to "ffmpeg" (system PATH) if bundled not found
        assert extractor.ffmpeg_path == "ffmpeg"

    def test_yt_dlp_resolves_from_system_path(self):
        """VideoReferenceAnalyzer falls back to system yt-dlp when bundled not found."""
        from nyrahost.tools.video_reference_analyzer import VideoReferenceAnalyzer

        analyzer = VideoReferenceAnalyzer(claude_backend=MagicMock())
        yt_path = analyzer._resolve_yt_dlp_path()
        assert yt_path == "yt-dlp"  # Falls back to system PATH

    def test_error_messages_are_user_friendly(self):
        """Error messages don't expose internal error codes to users."""
        from nyrahost.tools.demo02_cli_tool import Demo02CLITool

        tool = Demo02CLITool(claude_backend=MagicMock())

        # Duration error
        with patch.object(tool, '_validate_file_duration', return_value=False):
            result = tool.execute({"source": "/test.mp4", "source_type": "file"})
            assert not result.is_ok
            assert "10" in result.error
            assert "DEMO-02" in result.error

    def test_confusion_pairs_defined(self):
        """Dolly vs truck confusion pairs are defined for user guidance."""
        from nyrahost.tools.shot_block_ui import ShotBlockConfirmationUI, CONFUSION_PAIRS
        from nyrahost.tools.video_llm_parser import CameraMoveType

        assert CameraMoveType.DOLLY in CONFUSION_PAIRS
        assert CameraMoveType.TRUCK in CONFUSION_PAIRS
        assert CONFUSION_PAIRS[CameraMoveType.DOLLY] == CameraMoveType.TRUCK
        assert CONFUSION_PAIRS[CameraMoveType.TRUCK] == CameraMoveType.DOLLY

    def test_all_demo02_tools_have_names(self):
        """All DEMO-02 tools have correct MCP tool names."""
        from nyrahost.tools.demo02_cli_tool import Demo02CLITool
        from nyrahost.tools.sequencer_tools import (
            SequencerCreateTool, SequencerAddCameraTool,
            SequencerSetKeyframeTool, SequencerAuthorShotTool,
        )
        from nyrahost.tools.video_reference_analyzer import VideoAnalysisTool

        expected_names = {
            "nyra_demo02",
            "nyra_sequencer_create",
            "nyra_sequencer_add_camera",
            "nyra_sequencer_set_keyframe",
            "nyra_sequencer_author_shot",
            "nyra_video_analyze",
        }

        tool_instances = [
            Demo02CLITool(claude_backend=MagicMock()),
            SequencerCreateTool(),
            SequencerAddCameraTool(),
            SequencerSetKeyframeTool(),
            SequencerAuthorShotTool(),
            VideoAnalysisTool(claude_backend=MagicMock()),
        ]

        actual_names = {t.name for t in tool_instances}
        for name in expected_names:
            assert name in actual_names, f"Tool {name} not found in registered tools"

    def test_cold_start_requires_no_meshy(self):
        """DEMO-02 doesn't require Meshy for the basic cold-start path.

        DEMO-02 uses the user's asset library first (ACT-03 asset search).
        Meshy is only called for missing hero assets — and DEMO-02 works
        with "found assets" mode if no Meshy is available.
        """
        from nyrahost.tools.demo02_orchestrator import Demo02Orchestrator
        from nyrahost.tools.video_llm_parser import CameraMoveType, VideoReferenceParams, ShotBlock

        orchestrator = Demo02Orchestrator(claude_backend=MagicMock())

        # Verify Meshy is not called in the orchestrator
        # (Demo02Orchestrator only uses: LightingAuthoringTool, Sequencer tools)
        assert not hasattr(orchestrator, '_meshy_tool')
        assert hasattr(orchestrator, '_lighting_tool')
        assert hasattr(orchestrator, '_sequencer_create')

    def test_max_clip_duration_is_10_seconds(self):
        """MAX_CLIP_DURATION is 10 seconds per ROADMAP SC#1."""
        from nyrahost.tools.video_reference_analyzer import MAX_CLIP_DURATION

        assert MAX_CLIP_DURATION == 10.0


if __name__ == "__main__":
    pytest.main([__file__, "-x", "-q"])
