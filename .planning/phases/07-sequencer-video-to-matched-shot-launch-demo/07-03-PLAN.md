"""nyrahost.tools.demo02_cli_tool — DEMO-02 MCP tool entry point.

Phase 7 Wave 3: User-facing MCP tool for the LAUNCH DEMO.
Wraps Demo02Orchestrator with:
- Duration validation (rejects >10s)
- Progress logging
- Ephemeral processing disclaimer
- User-friendly error messages
- Source type detection

Per ROADMAP SC#1: DEMO-02 ships end-to-end (no competitor has this as of April 2026).
Per ROADMAP SC#4: Ephemeral video processing — video deleted after analysis.
Per ROADMAP SC#5: First-install cold-start test — this tool is the entry point.
"""
from __future__ import annotations

import asyncio
import structlog
from typing import Any

from nyrahost.tools.base import NyraTool, NyraToolResult
from nyrahost.tools.demo02_orchestrator import Demo02Orchestrator
from nyrahost.tools.video_reference_analyzer import MAX_CLIP_DURATION

log = structlog.get_logger("nyrahost.tools.demo02_cli_tool")

__all__ = ["Demo02CLITool"]

# Error code prefix for DEMO-02: 32060-32099
ERR_DEMO02_DURATION = "[-32061]"
ERR_DEMO02_DURATION_MSG = (
    f"DEMO-02 requires clips ≤10 seconds. Your clip is longer. "
    f"Please provide a shorter clip or use the first 10 seconds of a longer video."
)


class Demo02CLITool(NyraTool):
    """MCP tool entry point for DEMO-02 LAUNCH DEMO.

    Tool name: nyra_demo02
    Alias for: nyra_demo02_run (Demo02Orchestrator's tool)

    User workflow:
      1. User pastes YouTube URL or attaches mp4 in NYRA chat
      2. User types "run demo02" or uses /demo02 slash command
      3. NYRA shows ephemeral disclaimer: "Video is processed temporarily and deleted after analysis"
      4. NYRA validates duration (rejects >10s with clear error)
      5. NYRA runs Demo02Orchestrator.analyze_and_assemble()
      6. If requires_confirmation: returns confirmation card
      7. If auto-commits: returns sequence_path + camera + lighting summary
    """
    name = "nyra_demo02"
    description = (
        "Run the LAUNCH DEMO: analyze a reference video (YouTube URL or mp4 ≤10s) and "
        "assemble a matching UE scene with Sequencer camera. "
        "Shows ephemeral processing disclaimer. "
        "Requires user confirmation if camera movement is unclear (dolly vs truck confusion). "
        "Creates: lighting setup (SCENE-01), level sequence, CineCamera, Sequencer keyframes. "
        "DEMO-02: the capability no competitor ships as of April 2026."
    )
    parameters = {
        "type": "object",
        "properties": {
            "source": {
                "type": "string",
                "description": (
                    "YouTube URL or absolute path to local mp4/mov file. "
                    f"Maximum duration: {MAX_CLIP_DURATION} seconds."
                ),
            },
            "source_type": {
                "type": "string",
                "enum": ["youtube_url", "file"],
                "default": "auto",
                "description": (
                    "'youtube_url' for YouTube links, 'file' for local files. "
                    "Default 'auto' detects from URL pattern."
                ),
            },
            "confirmed_shot_blocks": {
                "type": "array",
                "description": (
                    "User-confirmed shot blocks from the confirmation dialog. "
                    "Required if the previous call returned requires_confirmation=True."
                ),
            },
        },
        "required": ["source"],
    }

    EPHEMERAL_DISCLAIMER = (
        "DEMO-02 uses ephemeral video processing: your video is downloaded to a "
        "temporary location, keyframes are extracted for analysis, and the full "
        "video file is deleted after analysis. Only extracted keyframes are retained "
        "(no full video stored). This runs entirely on your machine."
    )

    def __init__(self, claude_backend: Any = None):
        super().__init__()
        self.claude_backend = claude_backend

    def execute(self, params: dict) -> NyraToolResult:
        source = params["source"]

        # Step 1: Detect source_type if "auto"
        source_type = params.get("source_type", "auto")
        if source_type == "auto":
            source_type = self._detect_source_type(source)

        # Step 2: Quick duration check (for local files, check immediately)
        if source_type == "file":
            if not self._validate_file_duration(source):
                return NyraToolResult.err(f"{ERR_DEMO02_DURATION} {ERR_DEMO02_DURATION_MSG}")

        # Step 3: Run the pipeline
        try:
            loop = asyncio.new_event_loop()
            result = loop.run_until_complete(
                self._run_pipeline(source, source_type, params.get("confirmed_shot_blocks"))
            )
            loop.close()

            # Add ephemeral disclaimer to message
            if result.get("requires_confirmation"):
                result["ephemeral_disclaimer"] = self.EPHEMERAL_DISCLAIMER
            else:
                result["ephemeral_disclaimer"] = self.EPHEMERAL_DISCLAIMER

            return NyraToolResult.ok({
                **result,
                "source_type": source_type,
                "ephemeral_disclaimer": self.EPHEMERAL_DISCLAIMER,
            })

        except ValueError as e:
            if "exceeds maximum" in str(e) or "10" in str(e):
                return NyraToolResult.err(f"{ERR_DEMO02_DURATION} {ERR_DEMO02_DURATION_MSG}")
            return NyraToolResult.err(str(e))
        except RuntimeError as e:
            err_str = str(e)
            # Translate error codes to user-friendly messages
            if "32037" in err_str:
                return NyraToolResult.err(
                    f"DEMO-02 download failed: Could not download the YouTube video. "
                    f"Check the URL and your internet connection. Details: {err_str}"
                )
            if "32036" in err_str:
                return NyraToolResult.err(
                    f"DEMO-02 extraction failed: Could not extract frames from the video. "
                    f"Ensure ffmpeg is installed and the video is valid. Details: {err_str}"
                )
            if "32040" in err_str:
                return NyraToolResult.err(
                    f"DEMO-02 analysis failed: Claude vision analysis encountered an error. "
                    f"Try again or use a different clip. Details: {err_str}"
                )
            return NyraToolResult.err(f"DEMO-02 failed: {err_str}")
        except Exception as e:
            log.error("demo02_cli_unexpected_error", error=str(e))
            return NyraToolResult.err(f"[-32069] Unexpected error in DEMO-02: {e}")

    def _detect_source_type(self, source: str) -> str:
        """Detect whether source is a YouTube URL or local file."""
        if any(host in source.lower() for host in ["youtube.com", "youtu.be", "youtube.be"]):
            return "youtube_url"
        return "file"

    def _validate_file_duration(self, file_path: str) -> bool:
        """Check if a local file is within the duration limit.

        Returns True if valid, False if too long or file not found.
        """
        from pathlib import Path
        from nyrahost.tools.ffmpeg_extractor import FFmpegKeyframeExtractor

        if not Path(file_path).exists():
            # Let the orchestrator handle the file-not-found error
            return True

        try:
            extractor = FFmpegKeyframeExtractor()
            duration = extractor._get_video_duration(file_path)
            return duration <= MAX_CLIP_DURATION
        except Exception:
            # Let the orchestrator handle extraction errors
            return True

    async def _run_pipeline(
        self, source: str, source_type: str, confirmed_shot_blocks: list | None
    ) -> dict[str, Any]:
        """Run Demo02Orchestrator pipeline with progress logging."""
        orchestrator = Demo02Orchestrator(claude_backend=self.claude_backend)

        log.info("demo02_start", source=source, source_type=source_type)
        result = await orchestrator.analyze_and_assemble(
            source=source,
            source_type=source_type,
            confirmed_shot_blocks=confirmed_shot_blocks,
        )
        log.info("demo02_pipeline_complete",
                 requires_confirmation=result.get("requires_confirmation"))
        return result
