class CanaryVerifier:
    """Verifies DEMO-01 assembly output against success criteria."""

    def __init__(
        self,
        min_actors: int = 5,
        max_actors: int = 20,
        require_library_preference: bool = True,
    ):
        self.min_actors = min_actors
        self.max_actors = max_actors
        self.require_library_preference = require_library_preference

    def verify(self, result: AssemblyResult) -> tuple[int, str]:
        """Verify AssemblyResult against success criteria.

        Returns (exit_code, message):
          0 = PASS
          1 = FAIL
          2 = PARTIAL
        """
        # SC#3: 5-20 actors placed
        actor_count = len(result.placed_actors)
        if not (self.min_actors <= actor_count <= self.max_actors):
            return 1, f"FAIL: Expected {self.min_actors}-{self.max_actors} actors, got {actor_count}"

        # SC#3: at least one material applied
        material_count = len(result.applied_materials)
        if material_count < 1:
            return 1, f"FAIL: Expected at least 1 material applied, got {material_count}"

        # SC#3: at least one lighting actor configured
        lighting_count = len(result.lighting_actors)
        if lighting_count < 1:
            return 1, f"FAIL: Expected at least 1 lighting actor, got {lighting_count}"

        # SC#1: library preference — at least one actor from library
        if self.require_library_preference:
            lib_actors = [
                a for a in result.placed_actors
                if a.get("source", "unknown") == "library"
            ]
            if len(lib_actors) == 0:
                # PARTIAL is acceptable — Meshy/ComfyUI might have generated all
                return 2, "PARTIAL: No library actors (all generated — acceptable if external services available)"

        # All criteria met
        return 0, (
            f"PASS: {actor_count} actors placed, {material_count} materials applied, "
            f"{lighting_count} lighting actors configured."
        )

def test_cold_start_reliability() -> tuple[int, str]:
    """Verify Meshy unavailable -> ComfyUI fallback -> placeholder material never blocks.

    Sets MESHY_API_KEY to invalid value, runs resolution through AssetFallbackChain,
    verifies it returns a placeholder without blocking.
    """
    log.info("canary_cold_start_test")

    original_key = os.environ.get("MESHY_API_KEY")
    original_host = os.environ.get("COMFYUI_HOST")

    try:
        # Block Meshy
        os.environ["MESHY_API_KEY"] = "invalid_key_for_testing"
        os.environ["COMFYUI_HOST"] = "127.0.0.1:99999"  # unreachable

        chain = AssetFallbackChain()
        result = chain.resolve_actor_asset("test_prop_xyz_not_in_library", "hero_prop")

        # Must return a placeholder, never block
        if result.source == "placeholder":
            return 0, "PASS: Cold-start reliability verified — Meshy unavailable -> placeholder (never blocked)"
        else:
            return 1, f"FAIL: Expected placeholder fallback, got source={result.source}"

    finally:
        # Restore
        if original_key:
            os.environ["MESHY_API_KEY"] = original_key
        if original_host:
            os.environ["COMFYUI_HOST"] = original_host

# Main canary run
# ---------------------------------------------------------------------------

def run_canary(
    test_image: Optional[str] = None,
    random_image: bool = False,
    expect_actors: int = 5,
) -> int:
    """Run the DEMO-01 canary test.

    Args:
        test_image: Path to a specific test image (optional)
        random_image: If True, pick a random image from bundled set
        expect_actors: Minimum expected actor count (default 5)

    Returns:
        Exit code: 0 (PASS), 1 (FAIL), 2 (PARTIAL)
    """
    log.info("canary_run_started", test_image=test_image, random_image=random_image)

    # Resolve test image
    if random_image:
        name = random.choice(BUNDLED_TEST_IMAGES)
        image_path = resolve_test_image(name)
    elif test_image:
        image_path = Path(test_image)
    else:
        name = BUNDLED_TEST_IMAGES[0]
        image_path = resolve_test_image(name)

    if not image_path.exists():
        log.error("canary_image_not_found", path=str(image_path))
        return 1

    log.info("canary_using_image", path=str(image_path), name=name if 'name' in dir() else "custom")

    # Run cold-start reliability test first
    cold_result = test_cold_start_reliability()
    if cold_result != 0:
        log.warning("canary_cold_start_failed", code=cold_result)

    # Run the assembly
    tool = AssembleSceneTool()
    start = time.monotonic()

    result = tool.execute({
        "reference_image_path": str(image_path),
        "dry_run": False,
    })

    elapsed = time.monotonic() - start
    log.info("canary_assembly_complete", elapsed=f"{elapsed:.1f}s", result=result.error or "ok")

    if result.error:
        # Check for acceptable errors (external service unavailable, etc.)
        if any(x in result.error for x in ["MESHY_API_KEY", "COMFYUI", "API key", "connection refused"]):
            log.warning("canary_external_service_unavailable", error=result.error)
            return 2  # PARTIAL — acceptable degraded
        return 1  # FAIL — unexpected error

    # Parse result into AssemblyResult
    assembly_result = _result_to_assembly_result(result)

    # Verify
    verifier = CanaryVerifier(min_actors=expect_actors)
    exit_code, message = verifier.verify(assembly_result)
    log.info("canary_verdict", exit_code=exit_code, message=message)

    # Print structured result for log
    print(f"\n{'='*60}")
    print(f"NYCANA Canary Result: {message}")
    print(f"Elapsed: {elapsed:.1f}s")
    if assembly_result.log_entries:
        print(f"Log entries ({len(assembly_result.log_entries)}):")
        for entry in assembly_result.log_entries[:5]:  # first 5
            print(f"  - {entry}")
        if len(assembly_result.log_entries) > 5:
            print(f"  ... and {len(assembly_result.log_entries) - 5} more")
    print(f"{'='*60}\n")

    return exit_code


def _result_to_assembly_result(result) -> AssemblyResult:
    """Convert NyraToolResult to AssemblyResult for verification."""
    if not result.is_ok:
        return AssemblyResult(success=False, error_message=result.error or "Unknown error")

    data = result.data or {}
    placed = data.get("actors_placed", [])
    materials = data.get("materials_applied", [])
    lighting = data.get("lighting_configured", [])
    log_entries = data.get("log_entries", [])

    return AssemblyResult(
        placed_actors=placed if isinstance(placed, list) else [],
        applied_materials=materials if isinstance(materials, list) else [],
        lighting_actors=lighting if isinstance(lighting, list) else [],
        log_entries=log_entries if isinstance(log_entries, list) else [],
        success=data.get("success", True),
        error_message=data.get("error_message"),
    )
