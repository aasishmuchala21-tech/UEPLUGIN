## Purpose

Phase 5 exit gate validates that all external tool integrations (Meshy, ComfyUI, computer-use) are registered in the NyraToolCatalog, wired to NyraHost, and pass operator-level smoke tests. This plan does NOT implement new tools — it produces the validation harness and final verdict.

// === PHASE 5: External Tool Integrations ===
UE_LOG(LogNyraDev, Display, TEXT("=== Phase 5: External Tool Integrations ==="));

// GEN-01: Meshy REST
TArray<FString> Phase5MeshyTools = {
    TEXT("nyra_meshy_image_to_3d"),
    TEXT("nyra_job_status")
};
for (const auto& ToolName : Phase5MeshyTools) {
    if (UNyraDevTools* DevTools = UNyraDevTools::Get()) {
        bool bRegistered = DevTools->IsToolRegistered(ToolName);
        UE_LOG(LogNyraDev, Display, TEXT("  [%s] %s"),
               bRegistered ? TEXT("OK") : TEXT("MISSING"),
               *ToolName);
        if (!bRegistered) {
            MissingTools.Add(ToolName);
        }
    }
}
